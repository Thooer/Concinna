## **根目录结构**


```
/ (Project Root)
├── .clang-format                   # [代码规范] Clang 代码格式化配置文件
├── .gitignore                      # [版本控制] Git 忽略文件配置
├── .cicd/                          # [工程] CI/CD 自动化配置
│   ├── workflows/                  #   - GitHub Actions 工作流定义
│   │   ├── build.yml               #     - 多平台构建工作流，支持并行构建
│   │   ├── test.yml                #     - 自动化测试工作流，包含单元测试和集成测试
│   │   └── release.yml             #     - 发布工作流，自动打包和分发
│   ├── scripts/                    #   - CI/CD 辅助脚本
│   │   ├── setup-environment.sh    #     - 构建环境初始化脚本
│   │   └── deploy.sh               #     - 部署脚本
│   └── configs/                    #   - CI/CD 配置文件
│       ├── build-matrix.json       #     - 构建矩阵配置（平台、编译器组合）
│       └── test-config.json        #     - 测试配置参数
├── .githooks/                      # [工程] Git 钩子脚本
│   ├── pre-commit                  #   - 提交前代码质量检查（格式化、静态分析）
│   ├── pre-push                    #   - 推送前自动测试运行
│   └── commit-msg                  #   - 提交信息格式检查（遵循约定式提交）
│
├── Binaries/                       # [构建产物] 最终的可执行文件、库文件 (被.gitignore)
│   ├── Win64/
│   ├── Linux/
│   └── ...
├── Intermediate/                   # [构建产物] 编译中间文件和自动生成的代码 (被.gitignore)
│   ├── Build/                      #   - .obj, .o 等编译中间文件
│   └── Generated/                  #   - 反射系统、构建系统等生成的代码
│
├── Build/                          # [工程] 构建系统元配置
│   ├── CMake/                      #   - 自定义 CMake 模块和函数
│   │   ├── Modules/                #     - 查找第三方库的 Find*.cmake 模块
│   │   ├── Functions/              #     - 通用构建函数 (如模块注册、测试配置)
│   │   └── Vcpkg/                  #     - vcpkg 集成配置文件
│   ├── Configs/                    #   - 不同构建类型的配置文件
│   │   ├── Debug.cmake             #     - Debug 构建配置
│   │   ├── Release.cmake           #     - Release 构建配置
│   │   ├── RelWithDebInfo.cmake    #     - 带调试信息的发布版本
│   │   └── MinSizeRel.cmake        #     - 最小体积发布版本
│   ├── Toolchains/                 #   - 交叉编译工具链文件
│   │   ├── Windows-MSVC.cmake      #     - Windows MSVC 工具链
│   │   ├── Windows-Clang.cmake     #     - Windows Clang 工具链
│   │   └── Linux-GCC.cmake         #     - Linux GCC 工具链
│   ├── Templates/                  #   - 模块 CMakeLists.txt 模板
│   └── Scripts/                    #   - 构建辅助脚本
│       ├── GenerateProjects.cmake  #     - 项目文件生成脚本
│       └── PackageDistribution.cmake #   - 打包分发脚本
│
├── Engine/                         # [核心支柱] 引擎本身 (遵循模块元信息标准)
│   ├── Binaries/                   #   - 引擎模块编译产物 (非最终可执行文件)
│   ├── Config/                     #   [新增] 引擎层级的默认配置文件 (DefaultEngine.ini)
│   ├── Content/                    #   - 引擎内置的、已处理过的资源 (默认材质、字体、图标)
│   ├── Plugins/                    #   [调整] 核心插件。引擎功能的核心组成部分，默认启用，与引擎一同分发
│   │   ├── Runtime/                #     - 运行时功能的插件 (如新物理引擎集成)
│   │   └── Developer/              #     - 开发/编辑器功能的插件 (如新版本控制系统集成)
│   ├── Shaders/                    #   - 引擎内置的着色器源码 (.hlsl, .glsl)
│   └── Source/                     #   - 引擎C++源码 (详细结构见下文)
│
├── Projects/                       # [新增] 所有“项目”的集合 (取代了Games)
│   └── SampleGame/                 #   - 示例游戏项目 (一个独立、自包含的单元)
│       ├── Binaries/               #       - 游戏专属代码的编译产物
│       ├── Build/                  #       - [可选] 游戏专属的构建脚本
│       ├── Config/                 #       - 游戏专属配置 (继承并覆盖 Engine/Config)
│       ├── Content/                #       - 游戏资源
│       ├── Plugins/                #       - [新增] 项目插件。与特定游戏逻辑或内容紧密耦合。
│       │   └── AuctionHouse/       #         - 仅用于SampleGame的拍卖行插件
│       ├── Shaders/
│       ├── Source/                 #       - 游戏逻辑C++代码
│       └── SampleGame.project      #       [核心新增] 项目描述符文件 (JSON/XML)
│                                   #           - 定义项目名、引擎依赖版本、启用的引擎插件、
│                                   #             包含的项目插件列表等。
│
├── Plugins/                        # [核心支柱] 市场/可选插件。独立的、可插拔的功能模块，项目可以按需选择性地启用。例如，一个高级的海洋渲染插件、一个特定的语音聊天集成插件。
│   └── ExamplePlugin/              #   - 示例插件 (遵循模块元信息标准)
│       ├── Content/                #       - 插件提供的资源
│       ├── Shaders/                #       - 插件自定义的着色器
│       └── Source/                 #       - 插件C++源码
│
├── ThirdParty/                     # [依赖] 第三方依赖库 (源码或预编译)
│
├── Tools/                          # [支持] 独立于引擎运行时的工具链应用 
│   ├── AssetProcessor/             #   - 资源处理与转换工具
│   ├── BuildAndDeploy/             #   - 构建与发布自动化工具
│   ├── DebugAndProfile/            #   - 独立的调试与性能分析工具
│   ├── CodeGeneration/             #   - 代码生成工具
│   ├── QualityAssurance/           #   - 代码质量保证工具
│   └── Development/                #   - 开发辅助工具
│
├── ContentSource/                  # [源内容] 引擎、游戏和插件的原始美术资源 (未处理)
│   ├── Engine/                     #   - 引擎自带的源资源 (.psd, .fbx, .blend)
│   ├── Projects/
│   │   └── SampleGame/             #   - 示例游戏的源资源
│   └── Plugins/
│       └── ExamplePlugin/          #   - 示例插件的源资源
│
├── Docs/                           # [文档] 项目级文档
│
├── Scripts/                        # [开发] 开发和构建脚本
│   ├── Setup/                      #   - 环境搭建脚本
│   ├── Build/                      #   - 构建相关脚本
│   ├── Testing/                    #   - 测试相关脚本
│   ├── Maintenance/                #   - 维护脚本
│   └── Utilities/                  #   - 通用工具脚本
│
├── Legal/                          # [法律] 法律与合规文件
│   ├── EULA.txt                    #   - 最终用户许可协议
│   ├── ThirdPartyNotices.txt       #   - 第三方库许可声明
│   └── PRIVACY.md                  #   - 隐私政策
│
├── CMakeLists.txt                  # [根构建] 项目总构建脚本
├── CMakePresets.json               # [构建预设] CMake 预设配置文件
├── vcpkg.json                      # [依赖管理] vcpkg 包管理配置
└── README.md                       

```
